{{- $input := .Get 0 -}} {{- $id := "" -}} {{- /* Extract video ID from various YouTube URL formats */ -}} {{- if strings.Contains $input "youtube.com/watch" -}} {{- $id = (index (last 1 (split (index (split $input "v=") 1) "&")) 0) -}} {{- else if strings.Contains $input "youtu.be/" -}} {{- $id = (index (last 1 (split $input "youtu.be/")) 0) -}} {{- $id = (index (split $id "?") 0) -}} {{- else if strings.Contains $input "youtube.com/embed/" -}} {{- $id = (index (last 1 (split $input "embed/")) 0) -}} {{- $id = (index (split $id "?") 0) -}} {{- else -}} {{- /* Assume it's already a video ID */ -}} {{- $id = $input -}} {{- end -}} {{- $title := .Get "title" | default "YouTube video player" -}}
